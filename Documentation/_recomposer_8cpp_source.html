50  void Recomposer::setFisheye(const double fisheye)
51  {
52  double factor = 1. - clip_minmax(fisheye, 0., 1.);
53  for(unsigned int i = 0; i < m_number_of_channels; i++)
54  {
55  double azimuth = (double)i / (double)m_number_of_channels * HOA_2PI;
56  if(azimuth < HOA_PI)
57  azimuth *= factor;
58  else
59  azimuth = HOA_2PI - ((HOA_2PI - azimuth) * factor);
60  m_encoders[i]->setAzimuth(azimuth);
61  }
62  }
