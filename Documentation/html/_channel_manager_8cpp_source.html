138  void ChannelManager::rotateSelectedChannels(double newRadian, int channelBeingDragged, int magnet)
139  {
140  if (!isInside(channelBeingDragged, 0, m_channels.size()))
141  return;
142 
143  double deltaAngle;
144  double oldAngle = m_channels[channelBeingDragged]->getAzimuth();
145  deltaAngle = newRadian - m_channels[channelBeingDragged]->getAzimuth();
146  m_channels[channelBeingDragged]->setAzimuth(newRadian);
147 
148  if (magnet > 0)
149  {
150  if ( getClosestDefChannelDistance(channelBeingDragged) < HOA_2PI / (double) m_channels.size() *0.1 )
151  {
152  m_channels[channelBeingDragged]->setAzimuth(getClosestDefChannelAzimuth(channelBeingDragged));
153  deltaAngle = m_channels[channelBeingDragged]->getAzimuth() - oldAngle;
154  }
155  }
156 
157  for (int i = 0; i < m_channels.size(); i++)
158  {
159  if (isSelected(i) && i != channelBeingDragged)
160  {
161  m_channels[i]->rotateAzimuth(deltaAngle);
162  }

241  double ChannelManager::radianInterp(double step, double startRad, double endRad)
242  {
243  double start = wrap_twopi(startRad);
244  double end = wrap_twopi(endRad);
245 
246  // anti-clockwise
247 
248  if ( wrap_twopi(end - start) <= HOA_PI )
249  {
250  if (end - start >= 0)
251  return wrap_twopi( start + step*(end - start) );
252  else
253  return wrap_twopi( start + step*( (end + HOA_2PI) - start) );
254  }
255 
256  // clockwise
257 
258  else
259  {
260  if (end - start <= 0)
261  return wrap_twopi( start + step*(end - start) );
262  else
263  return wrap_twopi( start - step*( (start + HOA_2PI) - end) );
264  }
265  }

272  double ChannelManager::getClosestDefChannelAzimuth(double azimuth)
273  {
274  double angle = wrap_twopi(azimuth);
275  double distance = HOA_2PI;
276  double tempDistance, closestAngle;
277  closestAngle = angle;
278 
279  for (int i = 0; i < m_channels.size(); i++)
280  {
281  tempDistance = radianClosestDistance(angle, m_defaultAzimuths[i]);
282  if (tempDistance < distance) {
283  distance = tempDistance;
284  closestAngle = m_defaultAzimuths[i];
285  }
286  }
