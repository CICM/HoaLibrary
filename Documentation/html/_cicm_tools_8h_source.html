364  static double radianInterp(double _step, double _startRad, double _endRad)
365  {
366  _startRad = radian_wrap(_startRad);
367  _endRad = radian_wrap(_endRad);
368 
369  if ( radian_wrap(_endRad - _startRad) <= CICM_PI ) // anti-clockwise
370  {
371  if (_endRad - _startRad >= 0)
372  return radian_wrap( _startRad + _step*(_endRad - _startRad) );

379  return radian_wrap( _startRad + _step*(_endRad - _startRad) );

512  static double wrap(double _val, double _lo, double _hi){
513  double lo;
514  double hi;
515  if(_lo == _hi) return _lo;
516  if (_lo > _hi) {
517  hi = _lo; lo = _hi;
518  } else {
519  lo = _lo; hi = _hi;
520  }
521  const double range = hi - lo;
522  if (_val >= lo && _val < hi) return _val;
523  if (range <= 0.000000001) return lo; // no point...
524  const long numWraps = long((_val-lo)/range) - (_val < lo);
525  return _val - range * double(numWraps);
526  }
