51 
52  // Copy Harmonics Args(0)
53  for(unsigned int i = 0, j = 0; i < m_number_of_harmonics; i += i * 2 + 1, j++)
54  {
55  outputs[i] = inputs[i];
56  }
57 
58  tcos_x = cos_x = m_cos_yaw;
59  sin_x = m_sin_yaw;
60  // Perform Yaw Rotation
61  for(unsigned int i = 1; i <= m_order; i++)
62  {
63  for(unsigned int j = (i + 1) * (i + 1) - 1, k = i; j < m_number_of_harmonics; k++, j += k * 2 + 1)
64  {
65  sig = inputs[j-1];
66  outputs[j-1] = sin_x * inputs[j] + cos_x * sig;
67  outputs[j] = cos_x * inputs[j] - sin_x * sig;
68  }
69  cos_x = tcos_x * m_cos_yaw - sin_x * m_sin_yaw; // cos(x + b) = cos(x) * cos(b) - sin(x) * sin(b)
70  sin_x = tcos_x * m_sin_yaw + sin_x * m_cos_yaw; // sin(x + b) = cos(x) * sin(b) + sin(x) * cos(b)
71  tcos_x = cos_x;
72  }
