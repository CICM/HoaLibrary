45  void KitSources::PolarLines::setAzimuth(unsigned int index, double azimuth)
46  {
47  assert(index < m_number_of_sources);
48  m_values_new[index + m_number_of_sources] = wrap_twopi(azimuth);
49  m_values_old[index + m_number_of_sources] = wrap_twopi(m_values_old[index + m_number_of_sources]);
50  double distance;
51  if(m_values_old[index + m_number_of_sources] > m_values_new[index + m_number_of_sources])
52  distance = (m_values_old[index + m_number_of_sources] - m_values_new[index + m_number_of_sources]);
53  else
54  distance = (m_values_new[index + m_number_of_sources] - m_values_old[index + m_number_of_sources]);
55  if(distance <= HOA_PI)
56  {
57  m_values_step[index + m_number_of_sources] = (m_values_new[index + m_number_of_sources] - m_values_old[index + m_number_of_sources]) / (double)m_ramp;
58  }
59  else
60  {
61  if(m_values_new > m_values_old)
62  {
63  m_values_step[index + m_number_of_sources] = ((m_values_new[index + m_number_of_sources] - HOA_2PI) - m_values_old[index + m_number_of_sources]) / (double)m_ramp;
64  }
65  else
66  {
67  m_values_step[index + m_number_of_sources] = ((m_values_new[index + m_number_of_sources] + HOA_2PI) - m_values_old[index + m_number_of_sources]) / (double)m_ramp;
68  }
69  }
70  m_counter = 0;
71  }

89  void KitSources::PolarLines::process(float* vector)
90  {
91  cblas_saxpy(m_number_of_sources * 2, 1., m_values_step, 1, m_values_old, 1);
92  if(m_counter++ >= m_ramp)
93  {
94  cblas_scopy(m_number_of_sources * 2, m_values_new, 1, m_values_old, 1);
95  memset(m_values_step, 0, sizeof(double) * m_number_of_sources * 2);
96  m_counter = 0;
97  }
98  cblas_scopy(m_number_of_sources * 2, m_values_old, 1, vector, 1);
99  }

195  void KitSources::process(const float** ins, float** outs)
196  {
197  int numins = m_map->getNumberOfSources();
198  int numouts = m_decoder->getNumberOfChannels();
199  int nharmo = m_map->getOrder() * 2 + 1;
200  for(int i = 0; i < numins; i++)
201  {
202  cblas_scopy(m_vector_size, ins[i], 1, m_inputs_float+i, numins);
203  }
204 
205  if(m_decoder->getDecodingMode() == DecoderMulti::Regular)
206  {
207  for(int i = 0; i < m_vector_size; i++)
208  {
209  m_lines->process(m_lines_vector);
210  for(int j = 0; j < numins; j++)
211  m_map->setRadius(j, m_lines_vector[j]);
212  for(int j = 0; j < numins; j++)
213  m_map->setAzimuth(j, m_lines_vector[j+numins]);
214 
215  m_map->process(m_inputs_float + numins * i, m_harmonics_float + nharmo * i);
216  m_optim->process(m_harmonics_float + nharmo * i, m_harmonics_float + nharmo * i);
217  m_decoder->processRegular( m_harmonics_float + nharmo * i, m_outputs_float + numouts * i);
218  m_meter->process(m_outputs_float + numouts * i);
219  }
220  }
221  else if(m_decoder->getDecodingMode() == DecoderMulti::Irregular)
222  {
223  for(int i = 0; i < m_vector_size; i++)
224  {
225  m_lines->process(m_lines_vector);
226  for(int j = 0; j < numins; j++)
227  m_map->setRadius(j, m_lines_vector[j]);
228  for(int j = 0; j < numins; j++)
229  m_map->setAzimuth(j, m_lines_vector[j+numins]);
230  m_map->process(m_inputs_float + numins * i, m_harmonics_float + nharmo * i);
231  m_optim->process(m_harmonics_float + nharmo * i, m_harmonics_float + nharmo * i);
232  m_decoder->processIrregular(m_harmonics_float + nharmo * i, m_outputs_float + numouts * i);
233  m_meter->process(m_outputs_float + numouts * i);
234  }
235  }
236  else
237  {
238  for(int i = 0; i < m_vector_size; i++)
239  {
240  m_lines->process(m_lines_vector);
241  for(int j = 0; j < numins; j++)
242  m_map->setRadius(j, m_lines_vector[j]);
243  for(int j = 0; j < numins; j++)
244  m_map->setAzimuth(j, m_lines_vector[j+numins]);
245 
246  m_map->process(m_inputs_float + numins * i, m_harmonics_float + nharmo * i);
247  m_optim->process(m_harmonics_float + nharmo * i, m_harmonics_float + nharmo * i);
248  }
249  m_decoder->processBinaural((const float* const*)m_harmonics_float, (float **)m_outputs_float);
250  for(int i = 0; i < m_vector_size; i++)
251  {
252  m_meter->process(m_outputs_float + numouts * i);
253  }
254  }
255  for(int i = 0; i < numouts; i++)
256  {
257  cblas_scopy(m_vector_size, m_outputs_float+i, numouts, outs[i], 1);
258  }
