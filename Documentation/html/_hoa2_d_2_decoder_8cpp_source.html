82  void DecoderIrregular::setChannelPosition(unsigned int index, double azimuth)
83  {
84  long number_of_virutal_channels, channel_index1, channel_index2;
85  double current_distance, minimum_distance, angle, factor_index1, factor_index2, distance_index1, distance_index2;
86 
87  Planewaves::setChannelPosition(index, azimuth);
88 
89  // Sort the channels azimuth
90  memcpy(m_channels_azimuth_sorted, m_channels_azimuth, m_number_of_channels * sizeof(double));
91  std::sort(m_channels_azimuth_sorted, m_channels_azimuth_sorted+m_number_of_channels);
92 
93  // Get the minimum distance between the channels
94  minimum_distance = CICM_2PI + 1;
95  current_distance = distance_radian(m_channels_azimuth_sorted[0], m_channels_azimuth_sorted[m_number_of_channels-1]);
96  if(current_distance < minimum_distance)
97  minimum_distance = current_distance;
98  for(unsigned int i = 1; i < m_number_of_channels; i++)
99  {
100  current_distance = distance_radian(m_channels_azimuth_sorted[i], m_channels_azimuth_sorted[i-1]);
101  if(current_distance < minimum_distance)
102  minimum_distance = current_distance;
103  }
104 
105  // Get the optimal number of virtual channels
106  // Always prefer the minimum number of channels + 1
107  number_of_virutal_channels = (CICM_2PI / minimum_distance) + 1;
108  if(number_of_virutal_channels < m_number_of_harmonics + 1)
109  {
110  number_of_virutal_channels = m_number_of_harmonics + 1;
111  }
112 
113  // Compute the decoding matrix for sorted channels
114  for(unsigned int i = 0; i < number_of_virutal_channels; i++)
115  {
116  // Get the pair of real channels corresponding to the virtual channel
117  angle = (double)i / (double)number_of_virutal_channels * CICM_2PI;
118  minimum_distance = CICM_2PI + 1;
119  for(unsigned int j = 0; j < m_number_of_channels; j++)
120  {
121  current_distance = distance_radian(m_channels_azimuth_sorted[j], angle);
122  if(current_distance < minimum_distance)
123  {
124  minimum_distance = current_distance;
125  if(angle < m_channels_azimuth_sorted[j] && current_distance < CICM_2PI)
126  {
127  channel_index1 = j;
128  channel_index2 = j+1;
129  if(channel_index2 >= m_number_of_channels)
130  channel_index2 = 0;
131  }
132  else
133  {
134  channel_index1 = j;
135  channel_index2 = j-1;
136  if(channel_index2 < 0)
137  channel_index2 = m_number_of_channels - 1;
138  }
139  }
140  }
141 
142  // Get the factor for the pair of real channels
143  distance_index1 = distance_radian(m_channels_azimuth_sorted[channel_index1], angle);
144  distance_index2 = distance_radian(m_channels_azimuth_sorted[channel_index2], angle);
145  factor_index1 = cos(distance_index1 / (distance_index1 + distance_index2) * CICM_PI);
146  factor_index2 = cos(distance_index2 / (distance_index1 + distance_index2) * CICM_PI);
147 
148  // Get the harmonics coefficients for virtual channel
149  m_encoder->setAzimuth(angle);
150  m_encoder->process(1., m_harmonics_vector);
151 
152  m_decoder_matrix_sorted[channel_index1 * m_number_of_harmonics] = (0.5 / (double)(m_order + 1.)) * factor_index1;
153  m_decoder_matrix_sorted[channel_index2 * m_number_of_harmonics] = (0.5 / (double)(m_order + 1.)) * factor_index2;
154  for(unsigned int j = 1; j < m_number_of_harmonics; j++)
155  {
156  m_decoder_matrix_sorted[channel_index1 * m_number_of_harmonics + j] = (m_harmonics_vector[j] / (double)(m_order + 1.)) * factor_index1;
157  m_decoder_matrix_sorted[channel_index2 * m_number_of_harmonics + j] = (m_harmonics_vector[j] / (double)(m_order + 1.)) * factor_index2;
158  }
159  }
160 
161  // Copy the decoding matrix for sorted channels to the decoding matrix for unsorted channels
162  for(unsigned int i = 0; i < m_number_of_channels; i++)
163  {
164  for(unsigned int j = 0; j < m_number_of_channels; j++)
165  {
166  if(m_channels_azimuth[i] == m_channels_azimuth_sorted[j])
167  {
168  for(unsigned int k = 0; k < m_number_of_harmonics; k++)
169  {
170  m_decoder_matrix_float[i * m_number_of_harmonics + k] = m_decoder_matrix[i * m_number_of_harmonics + k] = m_decoder_matrix_sorted[j * m_number_of_harmonics + k];
171  }
172  }
173  }
174  }
