84  void DecoderIrregular::setChannelAzimuth(unsigned int index, double azimuth)
85  {
86  long number_of_virutal_channels;
87  double current_distance, minimum_distance;
88 
89  Planewaves::setChannelAzimuth(index, azimuth);
90 
91  // Get the minimum distance between the channels
92  minimum_distance = HOA_2PI + 1;
93  current_distance = distance_radian(m_channels_azimuth[0], m_channels_azimuth[m_number_of_channels-1]);
94  if(current_distance < minimum_distance)
95  minimum_distance = current_distance;
96  for(unsigned int i = 1; i < m_number_of_channels; i++)
97  {
98  current_distance = distance_radian(m_channels_azimuth[i], m_channels_azimuth[i-1]);
99  if(current_distance < minimum_distance)
100  minimum_distance = current_distance;
101  }
102 
103  // Get the optimal number of virtual channels
104  // Always prefer the number of harmonics + 2
105  number_of_virutal_channels = (HOA_2PI / minimum_distance);
106  if(number_of_virutal_channels < m_number_of_harmonics + 2)
107  {
108  number_of_virutal_channels = m_number_of_harmonics + 2;
109  }
110 
111  for(unsigned int i = 0; i < m_number_of_channels * m_number_of_harmonics; i++)
112  {
113  m_decoder_matrix[i] = 0.;
114  m_decoder_matrix_float[i] = 0.;
115  }
116 
117  // Compute the decoding matrix for sorted channels
118  for(unsigned int i = 0; i < number_of_virutal_channels; i++)
119  {
120  long channel_index1 = 0, channel_index2 = 0;
121  double factor_index1 = 0, factor_index2 = 0;
122 
123  // Get the pair of real channels corresponding to the virtual channel
124  double angle = (double)i / (double)number_of_virutal_channels * HOA_2PI;
125 
126  for(unsigned int j = 0; j < m_number_of_channels; j++)
127  {
128  if(j < m_number_of_channels-1 && angle >= m_channels_azimuth[j] && angle <= m_channels_azimuth[j+1])
129  {
130  channel_index1 = j;
131  channel_index2 = j+1;
132 
133  // Get the factor for the pair of real channels
134  double distance_index1 = angle - m_channels_azimuth[j];
135  double distance_index2 = m_channels_azimuth[j+1] - angle;
136  double distance_ratio = distance_index1 + distance_index2;
137  factor_index1 = cos(distance_index1 / (distance_ratio) * HOA_PI2);
138  factor_index2 = cos(distance_index2 / (distance_ratio) * HOA_PI2);
139  }
140  else if(angle >= m_channels_azimuth[m_number_of_channels-1] && angle <= m_channels_azimuth[0] + HOA_2PI)
141  {
142  channel_index1 = m_number_of_channels-1;
143  channel_index2 = 0;
144 
145  // Get the factor for the pair of real channels
146  double distance_index1 = angle - m_channels_azimuth[m_number_of_channels-1];
147  double distance_index2 = HOA_2PI - angle + m_channels_azimuth[0];
148  double distance_ratio = distance_index1 + distance_index2;
149  factor_index1 = cos(distance_index1 / (distance_ratio) * HOA_PI2);
150  factor_index2 = cos(distance_index2 / (distance_ratio) * HOA_PI2);
151  }
152  }
153 
154  // Get the harmonics coefficients for virtual channel
155  m_encoder->setAzimuth(angle + m_offset);
156  m_encoder->process(1., m_harmonics_vector);
157 
158  m_decoder_matrix[channel_index1 * m_number_of_harmonics] += (0.5 / (double)(m_order + 1.)) * factor_index1;
159  m_decoder_matrix[channel_index2 * m_number_of_harmonics] += (0.5 / (double)(m_order + 1.)) * factor_index2;
160  for(unsigned int j = 1; j < m_number_of_harmonics; j++)
161  {
162  m_decoder_matrix[channel_index1 * m_number_of_harmonics + j] += (m_harmonics_vector[j] / (double)(m_order + 1.)) * factor_index1;
163  m_decoder_matrix[channel_index2 * m_number_of_harmonics + j] += (m_harmonics_vector[j] / (double)(m_order + 1.)) * factor_index2;
164  }
165 
166  for(unsigned int i = 0; i < m_number_of_channels * m_number_of_harmonics; i++)
167  {
168  m_decoder_matrix_float[i] = m_decoder_matrix[i];
169  }
170  }
171  }

235  void DecoderBinaural::setSampleRate(unsigned int sampleRate)
236  {
237  int index;
238  float value_left;
239  float value_right;
240  if(sampleRate != m_sample_rate)
241  {
242  m_impulses_loaded = 0;
243  m_sample_rate = 0;
244  for(unsigned int i = 0; i < hoa_number_binaural_samplerate; i++)
245  {
246  if(sampleRate == hoa_binaural_samplerate[i])
247  {
248  index = i;
249  m_sample_rate = hoa_binaural_samplerate[i];
250  m_impulses_size = hoa_binaural_impulse_sizes[i];
251  }
252  }
253  if(!m_sample_rate)
254  {
255  index = 0;
256  m_sample_rate = hoa_binaural_samplerate[0];
257  m_impulses_size = hoa_binaural_impulse_sizes[0];
258  }
259  for(int i = 0; i < m_number_of_virtual_channels; i++)
260  {
261  m_impulses_vector[i] = get_mit_hrtf_2D(m_sample_rate, i * 360 / m_number_of_virtual_channels)+hoa_binaural_crop[index];
262  }
263 
264 
265  if(m_impulses_matrix)
266  delete [] m_impulses_matrix;
267 
268  m_impulses_matrix = new float[m_impulses_size * 2 * m_number_of_harmonics];
269 
270  for(unsigned int i = 0; i < m_number_of_harmonics; i++)
271  m_harmonics_vector[i] = 0.;
272  for(unsigned int i = 0; i < m_number_of_virtual_channels; i++)
273  m_channels_vector[i] = 0.;
274 
275  m_harmonics_vector[0] = 1.;
276  for(unsigned int i = 0; i < m_number_of_harmonics; i++)
277  {
278  if(i != 0)
279  {
280  m_harmonics_vector[i-1] = 0.;
281  m_harmonics_vector[i] = 1.;
282  }
283  m_decoder->process(m_harmonics_vector, m_channels_vector);
284 
285  for(unsigned int j = 0; j < m_impulses_size; j++)
286  {
287  value_left = 0;
288  value_right = 0;
289 
290  for(unsigned int k = 0; k < m_number_of_virtual_channels; k++)
291  {
292  value_left += m_channels_vector[k] * m_impulses_vector[k][j];
293  if(k == 0)
294  value_right += m_channels_vector[k] * m_impulses_vector[k][j];
295  else
296  value_right += m_channels_vector[k] * m_impulses_vector[m_number_of_virtual_channels - k][j];
297  }
298  m_impulses_matrix[j * m_number_of_harmonics + i] = value_left;
299  m_impulses_matrix[j * m_number_of_harmonics + i] *= (1. / 1.125) * (44100. / (double)m_sample_rate);
300  m_impulses_matrix[(j + m_impulses_size) * m_number_of_harmonics + i] = value_right;
301  m_impulses_matrix[(j + m_impulses_size) * m_number_of_harmonics + i] *= (1. / 1.125) * (44100. / (double)m_sample_rate);
302  }
303  }
304 
305  m_impulses_loaded = 1;
306  }
307  }

345  void DecoderBinaural::process(const float* const* inputs, float** outputs)
346  {
347  const float* input;
348  for(unsigned int i = 0; i < m_number_of_harmonics; i++)
349  {
350  input = inputs[i];
351  for(unsigned int j = 0; j < m_vector_size; j++)
352  {
353  m_input_matrix[i*m_vector_size+j] = input[j];
354  }
355  }
356  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, (m_impulses_size * 2), m_vector_size, m_number_of_harmonics, 1.,
357  m_impulses_matrix, m_number_of_harmonics,
358  m_input_matrix, m_vector_size,
359  0., m_result_matrix, m_vector_size);
360 
361  for(unsigned int j = 0; j < m_vector_size; j++)
362  {
363  cblas_saxpy(m_impulses_size,1.f, m_result_matrix+j+m_vector_size*m_impulses_size, m_vector_size, m_linear_vector_left + j, 1);
364  cblas_saxpy(m_impulses_size,1.f, m_result_matrix+j, m_vector_size, m_linear_vector_right + j, 1);
365 
366  outputs[0][j] = m_linear_vector_left[j];
367  outputs[1][j] = m_linear_vector_right[j];
368  }
369  cblas_scopy(m_impulses_size-1, m_linear_vector_left+m_vector_size, 1, m_linear_vector_left, 1);
370  cblas_scopy(m_impulses_size-1, m_linear_vector_right+m_vector_size, 1, m_linear_vector_right, 1);
371 
372  memset(m_linear_vector_left + m_impulses_size - 1, 0, m_vector_size * sizeof(float));
373  memset(m_linear_vector_right + m_impulses_size - 1, 0, m_vector_size * sizeof(float));
374  }
