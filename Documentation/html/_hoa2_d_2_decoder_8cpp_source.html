82  void DecoderIrregular::setChannelPosition(unsigned int index, double azimuth)
83  {
84  long number_of_virutal_channels, channel_index1, channel_index2;
85  double current_distance, minimum_distance, angle, factor_index1, factor_index2, distance_index1, distance_index2;
86 
87  Planewaves::setChannelPosition(index, azimuth);
88 
89  // Sort the channels azimuth
90  memcpy(m_channels_azimuth_sorted, m_channels_azimuth, m_number_of_channels * sizeof(double));
91  std::sort(m_channels_azimuth_sorted, m_channels_azimuth_sorted+m_number_of_channels);
92 
93  // Get the minimum distance between the channels
94  minimum_distance = HOA_2PI + 1;
95  current_distance = distance_radian(m_channels_azimuth_sorted[0], m_channels_azimuth_sorted[m_number_of_channels-1]);
96  if(current_distance < minimum_distance)
97  minimum_distance = current_distance;
98  for(unsigned int i = 1; i < m_number_of_channels; i++)
99  {
100  current_distance = distance_radian(m_channels_azimuth_sorted[i], m_channels_azimuth_sorted[i-1]);
101  if(current_distance < minimum_distance)
102  minimum_distance = current_distance;
103  }
104 
105  // Get the optimal number of virtual channels
106  // Always prefer the minimum number of channels + 1
107  number_of_virutal_channels = (HOA_2PI / minimum_distance) + 1;
108  if(number_of_virutal_channels < m_number_of_harmonics + 1)
109  {
110  number_of_virutal_channels = m_number_of_harmonics + 1;
111  }
112 
113  // Compute the decoding matrix for sorted channels
114  for(unsigned int i = 0; i < number_of_virutal_channels; i++)
115  {
116  // Get the pair of real channels corresponding to the virtual channel
117  angle = (double)i / (double)number_of_virutal_channels * HOA_2PI;
118  minimum_distance = HOA_2PI + 1;
119 
120  for(unsigned int j = 0; j < m_number_of_channels; j++)
121  {
122  current_distance = distance_radian(m_channels_azimuth_sorted[j], angle);
123  if(current_distance < minimum_distance)
124  {
125  minimum_distance = current_distance;
126  if(angle < m_channels_azimuth_sorted[j] && current_distance < HOA_PI)
127  {
128  channel_index1 = j;
129  channel_index2 = j+1;
130  }
131  else
132  {
133  channel_index1 = j;
134  channel_index2 = j-1;
135  }
136  }
137  }
138  channel_index2 = wrap(channel_index2, 0, m_number_of_channels - 1);
139  // Get the factor for the pair of real channels
140  distance_index1 = distance_radian(m_channels_azimuth_sorted[channel_index1], angle);
141  distance_index2 = distance_radian(m_channels_azimuth_sorted[channel_index2], angle);
142  factor_index1 = cos(distance_index1 / (distance_index1 + distance_index2) * HOA_PI);
143  factor_index2 = cos(distance_index2 / (distance_index1 + distance_index2) * HOA_PI);
144 
145  // Get the harmonics coefficients for virtual channel
146  m_encoder->setAzimuth(angle);
147  m_encoder->process(1., m_harmonics_vector);
148 
149  m_decoder_matrix_sorted[channel_index1 * m_number_of_harmonics] = (0.5 / (double)(m_order + 1.)) * factor_index1;
150  m_decoder_matrix_sorted[channel_index2 * m_number_of_harmonics] = (0.5 / (double)(m_order + 1.)) * factor_index2;
151  for(unsigned int j = 1; j < m_number_of_harmonics; j++)
152  {
153  m_decoder_matrix_sorted[channel_index1 * m_number_of_harmonics + j] = (m_harmonics_vector[j] / (double)(m_order + 1.)) * factor_index1;
154  m_decoder_matrix_sorted[channel_index2 * m_number_of_harmonics + j] = (m_harmonics_vector[j] / (double)(m_order + 1.)) * factor_index2;
155  }
156  }
157 
158  // Copy the decoding matrix for sorted channels to the decoding matrix for unsorted channels
159  for(unsigned int i = 0; i < m_number_of_channels; i++)
160  {
161  for(unsigned int j = 0; j < m_number_of_channels; j++)
162  {
163  if(m_channels_azimuth[i] == m_channels_azimuth_sorted[j])
164  {
165  for(unsigned int k = 0; k < m_number_of_harmonics; k++)
166  {
167  m_decoder_matrix_float[i * m_number_of_harmonics + k] = m_decoder_matrix[i * m_number_of_harmonics + k] = m_decoder_matrix_sorted[j * m_number_of_harmonics + k];
168  }
169  }
170  }
171  }

235  void DecoderBinaural::setSampleRate(unsigned int sampleRate)
236  {
237  int index;
238  float value_left;
239  float value_right;
240  if(sampleRate != m_sample_rate)
241  {
242  m_impulses_loaded = 0;
243  m_sample_rate = 0;
244  for(unsigned int i = 0; i < hoa_number_binaural_samplerate; i++)
245  {
246  if(sampleRate == hoa_binaural_samplerate[i])
247  {
248  index = i;
249  m_sample_rate = hoa_binaural_samplerate[i];
250  m_impulses_size = hoa_binaural_impulse_sizes[i];
251  }
252  }
253  if(!m_sample_rate)
254  {
255  index = 0;
256  m_sample_rate = hoa_binaural_samplerate[0];
257  m_impulses_size = hoa_binaural_impulse_sizes[0];
258  }
259  for(int i = 0; i < m_number_of_virtual_channels; i++)
260  {
261  m_impulses_vector[i] = get_mit_hrtf_2D(m_sample_rate, i * 360 / m_number_of_virtual_channels)+hoa_binaural_crop[index];
262  }
263 
264 
265  if(m_impulses_matrix)
266  delete [] m_impulses_matrix;
267 
268  m_impulses_matrix = new float[m_impulses_size * 2 * m_number_of_harmonics];
269 
270  for(unsigned int i = 0; i < m_number_of_harmonics; i++)
271  m_harmonics_vector[i] = 0.;
272  for(unsigned int i = 0; i < m_number_of_virtual_channels; i++)
273  m_channels_vector[i] = 0.;
274 
275  m_harmonics_vector[0] = 1.;
276  for(unsigned int i = 0; i < m_number_of_harmonics; i++)
277  {
278  if(i != 0)
279  {
280  m_harmonics_vector[i-1] = 0.;
281  m_harmonics_vector[i] = 1.;
282  }
283  m_decoder->process(m_harmonics_vector, m_channels_vector);
284 
285  for(unsigned int j = 0; j < m_impulses_size; j++)
286  {
287  value_left = 0;
288  value_right = 0;
289 
290  for(unsigned int k = 0; k < m_number_of_virtual_channels; k++)
291  {
292  value_left += m_channels_vector[k] * m_impulses_vector[k][j];
293  if(k == 0)
294  value_right += m_channels_vector[k] * m_impulses_vector[k][j];
295  else
296  value_right += m_channels_vector[k] * m_impulses_vector[m_number_of_virtual_channels - k][j];
297  }
298  m_impulses_matrix[j * m_number_of_harmonics + i] = value_left;
299  m_impulses_matrix[j * m_number_of_harmonics + i] *= (1. / 1.125) * (44100. / (double)m_sample_rate);
300  m_impulses_matrix[(j + m_impulses_size) * m_number_of_harmonics + i] = value_right;
301  m_impulses_matrix[(j + m_impulses_size) * m_number_of_harmonics + i] *= (1. / 1.125) * (44100. / (double)m_sample_rate);
302  }
303  }
304 
305  m_impulses_loaded = 1;
306  }

340  void DecoderBinaural::process(const float* const* inputs, float** outputs)
341  {
342  const float* input;
343  for(unsigned int i = 0; i < m_number_of_harmonics; i++)
344  {
345  input = inputs[i];
346  for(unsigned int j = 0; j < m_vector_size; j++)
347  {
348  m_input_matrix[i*m_vector_size+j] = input[j];
349  }
350  }
351  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, (m_impulses_size * 2), m_vector_size, m_number_of_harmonics, 1.,
352  m_impulses_matrix, m_number_of_harmonics,
353  m_input_matrix, m_vector_size,
354  0., m_result_matrix, m_vector_size);
355 
356  for(unsigned int j = 0; j < m_vector_size; j++)
357  {
358  cblas_saxpy(m_impulses_size,1.f, m_result_matrix+j+m_vector_size*m_impulses_size, m_vector_size, m_linear_vector_left + j, 1);
359  cblas_saxpy(m_impulses_size,1.f, m_result_matrix+j, m_vector_size, m_linear_vector_right + j, 1);

367  memset(m_linear_vector_left + m_impulses_size - 1, 0, m_vector_size * sizeof(float));
