39 void CicmLookAt(float eyeX, float eyeY, float eyeZ, float lookAtX, float lookAtY, float lookAtZ, float upX, float upY, float upZ)
40 {
41  // i am not using here proper implementation for vectors.
42  // if you want, you can replace the arrays with your own
43  // vector types
44  float f[3];
45 
46  // calculating the viewing vector
47  f[0] = lookAtX - eyeX;
48  f[1] = lookAtY - eyeY;
49  f[2] = lookAtZ - eyeZ;
50 
51 
52  float fMag, upMag;
53  fMag = sqrt(f[0]*f[0] + f[1]*f[1] + f[2]*f[2]);
54  upMag = sqrt(upX*upX + upY*upY + upZ*upZ);
55 
56  // normalizing the viewing vector
57  if( fMag != 0)
58  {
59  f[0] = f[0]/fMag;
60  f[1] = f[1]/fMag;
61  f[2] = f[2]/fMag;
62  }
63 
64  // normalising the up vector. no need for this here if you have your
65  // up vector already normalised, which is mostly the case.
66  if( upMag != 0 )
67  {
68  upX = upX/upMag;
69  upY = upY/upMag;
70  upZ = upZ/upMag;
71  }
72 
73  float s[3], u[3];
74 
75  CrossProd(f[0], f[1], f[2], upX, upY, upZ, s);
76  CrossProd(s[0], s[1], s[2], f[0], f[1], f[2], u);
77 
78  float M[]=
79  {
80  s[0], u[0], -f[0], 0,
81  s[1], u[1], -f[1], 0,
82  s[2], u[2], -f[2], 0,
83  0, 0, 0, 1
84  };
