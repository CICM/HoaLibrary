98  long SourcesPreset::interpolationSourceManager(SourcesManager* sourcesManagerSourceOne, SourcesManager* sourcesManagerSourceTwo, SourcesManager* sourcesManagerDestination, double aFrac)
99  {
100  double *color1, *color2;
101  sourcesManagerDestination->setExistence(0);
102  if(sourcesManagerSourceOne->getExistence() == 1 && sourcesManagerSourceTwo->getExistence() == 1)
103  {
104  sourcesManagerDestination->setExistence(1);
105  sourcesManagerDestination->setMaximumRadius(sourcesManagerSourceOne->getLimitMaximum());
106 
107  for(long i = 0; i <= sourcesManagerSourceOne->getMaximumIndexOfSource(); i++)
108  {
109  if(sourcesManagerSourceOne->sourceGetExistence(i) == 1 && sourcesManagerSourceTwo->sourceGetExistence(i) == 1)
110  {
111  color1 = sourcesManagerSourceOne->sourceGetColor(i);
112  color2 = sourcesManagerSourceTwo->sourceGetColor(i);
113  sourcesManagerDestination->sourceSetAbscissa(i, sourcesManagerSourceOne->sourceGetAbscissa(i) * (1 - aFrac)
114  + sourcesManagerSourceTwo->sourceGetAbscissa(i) * aFrac);
115  sourcesManagerDestination->sourceSetOrdinate(i, sourcesManagerSourceOne->sourceGetOrdinate(i) * (1 - aFrac)
116  + sourcesManagerSourceTwo->sourceGetOrdinate(i) * aFrac);
117  sourcesManagerDestination->sourceSetColor(i, color1[0] * (1. - aFrac) + color2[0] * aFrac,
118  color1[1] * (1. - aFrac) + color2[1] * aFrac,
119  color1[2] * (1. - aFrac) + color2[2] * aFrac,
120  color1[3] * (1. - aFrac) + color2[3] * aFrac);
121 
122  }
123  else if(sourcesManagerSourceOne->sourceGetExistence(i) == 1 && sourcesManagerSourceTwo->sourceGetExistence(i) == 0)
124  {
125  color1 = sourcesManagerSourceOne->sourceGetColor(i);
126 
127  sourcesManagerDestination->sourceSetRadius(i, sourcesManagerSourceOne->sourceGetRadius(i));
128  sourcesManagerDestination->sourceSetAzimuth(i, sourcesManagerSourceOne->sourceGetAzimuth(i));
129  sourcesManagerDestination->sourceSetColor(i, color1[0], color1[1], color1[2], color1[3]);
130  }
131  sourcesManagerDestination->sourceSetDescription(i, sourcesManagerSourceOne->sourceGetDescription(i));
132  sourcesManagerDestination->sourceSetMute(i, sourcesManagerSourceOne->sourceGetMute(i));
133  }
134  for(long i = 0; i <= sourcesManagerSourceOne->getMaximumIndexOfGroup(); i++)
135  {
136  if(sourcesManagerSourceOne->groupGetExistence(i) == 1)
137  {
138  for(long j = 0; j < sourcesManagerSourceOne->groupGetNumberOfSources(i); j++)
139  {
140  sourcesManagerDestination->groupSetSource(i, sourcesManagerSourceOne->groupGetSourceIndex(i, j));
141  }
142  color1 = sourcesManagerSourceOne->groupGetColor(i);
143  sourcesManagerDestination->groupSetColor(i, color1[0], color1[1], color1[2], color1[3]);
144  sourcesManagerDestination->groupSetDescription(i, sourcesManagerSourceOne->groupGetDescription(i));
145  }
146  if(sourcesManagerSourceOne->groupGetExistence(i) == 1 && sourcesManagerSourceTwo->groupGetExistence(i) == 1 )
147  {
148  color1 = sourcesManagerSourceOne->groupGetColor(i);
149  color2 = sourcesManagerSourceTwo->groupGetColor(i);
150  sourcesManagerDestination->groupSetColor(i, color1[0] * (1. - aFrac) + color2[0] * aFrac,
151  color1[1] * (1. - aFrac) + color2[1] * aFrac,
152  color1[2] * (1. - aFrac) + color2[2] * aFrac,
153  color1[3] * (1. - aFrac) + color2[3] * aFrac);
154  }
155  }
156  return 1;
157  }
158  return 0;
159  }

354  long SourcesPreset::recallFractionalSlot(SourcesManager* sourcesManager, double fractionnalIndex)
355  {
356  if(m_source_managers.size() == 0)
357  return 0;
358  else if(m_source_managers.size() == 1)
359  {
360  for(long i = 0; i < m_source_managers.size(); i++)
361  {
362  if(m_source_managers[i]->getExistence())
363  {
364  copySourceManager(m_source_managers[i], sourcesManager);
365  return 1;
366  }
367  }
368  }
369  else if(fractionnalIndex < 0.)
370  {
371  for(long i = 0; i < m_source_managers.size(); i++)
372  {
373  if(m_source_managers[i]->getExistence())
374  {
375  copySourceManager(m_source_managers[i], sourcesManager);
376  return 1;
377  }
378  }
379  }
380  else if(fractionnalIndex > m_source_managers.size() - 1)
381  {
382  for(long i = m_source_managers.size() - 1; i >= 0; i--)
383  {
384  if(m_source_managers[i]->getExistence())
385  {
386  copySourceManager(m_source_managers[i], sourcesManager);
387  return 1;
388  }
389  }
390  }
391  else
392  {
393  long indexOne = -1;
394  long indexTwo = -1;
395  for(long i = floor(fractionnalIndex); i >= 0 && indexOne == -1; i--)
396  {
397  if(m_source_managers[i]->getExistence())
398  indexOne = i;
399  }
400  if(indexOne == -1)
401  {
402  for(long i = 0; i < m_source_managers.size() && indexOne == -1; i++)
403  {
404  if(m_source_managers[i]->getExistence())
405  indexOne = i;
406  }
407  if(indexOne != -1)
408  {
409  copySourceManager(m_source_managers[indexOne], sourcesManager);
410  }
411  return 1;
412  }
413  else
414  {
415  for(long i = indexOne+1; i < m_source_managers.size() && indexTwo == -1; i++)
416  {
417  if(m_source_managers[i]->getExistence())
418  indexTwo = i;
419  }
420  if(indexTwo == -1)
421  {
422  copySourceManager(m_source_managers[indexOne], sourcesManager);
423  return 1;
424  }
425  else
426  {
427  double fracPart = clip_minmax((double)(fractionnalIndex - (long)indexOne) / (double)(indexTwo - indexOne), 0., 1.);
428  return interpolationSourceManager(m_source_managers[indexOne], m_source_managers[indexTwo], sourcesManager, fracPart);
429  }
430  }
431  }
432  return 0;
433  }
