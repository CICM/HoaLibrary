11  Ambisonic::Ambisonic(unsigned int order)
12  {
13  assert(order > 0);
14  m_order = order;
15  m_number_of_harmonics = (m_order + 1) * (m_order + 1);
16  m_number_of_inputs = m_number_of_harmonics;
17  m_number_of_outputs = m_number_of_harmonics;
18 
19  m_harmonics_arguments = new long[m_number_of_harmonics];
20  m_harmonics_bands = new long[m_number_of_harmonics];
21  for(int i = 0; i < m_number_of_harmonics; i++)
22  {
23  m_harmonics_bands[i] = sqrtf((float)i);
24  int index = i - (m_harmonics_bands[i]) * (m_harmonics_bands[i]);
25 
26  int index2 = (index - 1) / 2. + 1.;
27  if (index % 2 == 1)
28  index2 = -index2;
29  m_harmonics_arguments[i] = index2;
30  }
31  }
