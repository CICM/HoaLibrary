46  void SourcesGroup::computeCentroid()
47  {
48  m_centroid_x = 0.;
49  m_centroid_y = 0.;
50  if(m_sources.size())
51  {
52  for(int i = 0; i < m_sources.size(); i++)
53  {
54  if(m_source_manager->sourceGetExistence(m_sources[i]))
55  {
56  m_centroid_x += m_source_manager->sourceGetAbscissa(m_sources[i]);
57  m_centroid_y += m_source_manager->sourceGetOrdinate(m_sources[i]);
58  }
59  }
60  m_centroid_x /= m_sources.size();
61  m_centroid_y /= m_sources.size();
62  }

109  void SourcesGroup::shiftRadius(double radius)
110  {
111  if(m_maximum_radius >= 0)
112  {
113  if(radius < 0.)
114  {
115  double refRadius = m_maximum_radius;
116  for(int i = 0; i < m_sources.size(); i++)
117  {
118  if(m_source_manager->sourceGetRadius(m_sources[i]) < refRadius)
119  {
120  refRadius = m_source_manager->sourceGetRadius(m_sources[i]);
121  }
122  }
123  if(radius + refRadius < 0.)
124  {
125  radius = - refRadius;
126  }
127  }
128  else if(radius >= 0.)
129  {
130  double refRadius = -m_maximum_radius;
131  for(int i = 0; i < m_sources.size(); i++)
132  {
133  if(m_source_manager->sourceGetRadius(m_sources[i]) > refRadius)
134  {
135  refRadius = m_source_manager->sourceGetRadius(m_sources[i]);
136  }
137  }
138  if(radius + refRadius > m_maximum_radius)
139  {
140  radius = m_maximum_radius - refRadius;
141  }
142  }
143  }
144  for(int i = 0; i < m_sources.size(); i++)
145  {
146  m_source_manager->sourceSetRadius(m_sources[i], radius + m_source_manager->sourceGetRadius(m_sources[i]));
147  }
148  }
