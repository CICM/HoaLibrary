25  {
26  float cos_x = m_cosx;
27  float sin_x = m_sinx;
28  float tcos_x = cos_x;
29  outputs[0] = inputs[0];
30  for(int i = 2; i < m_number_of_harmonics; i += 2)
31  {
32  outputs[i] = cos_x * inputs[i] - sin_x * inputs[i-1];
33  outputs[i-1] = sin_x * inputs[i] + cos_x * inputs[i-1];
34  cos_x = tcos_x * m_cosx - sin_x * m_sinx; // cos(x + b) = cos(x) * cos(b) - sin(x) * sin(b)
35  sin_x = tcos_x * m_sinx + sin_x * m_cosx; // sin(x + b) = cos(x) * sin(b) + sin(x) * cos(b)
36  tcos_x = cos_x;
37  }
