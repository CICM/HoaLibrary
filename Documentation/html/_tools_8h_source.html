105  inline double associated_legendre(int l, int m, double x)
106  {
107  l = abs(l);
108  m = abs(m);
109 
110  if(l == m)
111  {
112  return pow(-1.0f, (double)m) * double_factorial(2. * m - 1) * pow(1. - x * x, 0.5 * m);
113  }
114  else if(l == m + 1)
115  {
116  return x * (2 * m + 1) * associated_legendre(m, m, x);
117  }
118  else
119  {
120  return ((double)(2 * l - 1) * x * associated_legendre(l - 1, m, x) - (double)(l + m - 1.) * associated_legendre(l - 2, m, x)) / (double)(l - m);
121  }
122  }
