105  inline double associated_legendre(int l, int m, double x)
106  {
107  l = abs(l);
108  m = abs(m);
109 
110  if(l == m)
111  {
112  return pow(-1.0f, (double)m) * pow(1. - x * x, 0.5 * m) * double_factorial(2. * m - 1);
113  }
114  else if(l == m + 1)
115  {
116  return x * associated_legendre(m, m, x) * (2 * m + 1);
117  }
118  else
119  {
120  return ((double)(2 * l - 1) * x * associated_legendre(l - 1, m, x) - (double)(l + m - 1.) * associated_legendre(l - 2, m, x)) / (double)(l - m);
121  }
122  }

406  inline void sphere_discretize(unsigned int numberOfPoints, double* azimuths, double* elevations)
407  {
408  if(numberOfPoints == 4) // Tethrahedron
409  {
410  azimuths[0] = 0.;
411  elevations[0] = HOA_PI2;
412  azimuths[1] = 0.;
413  azimuths[2] = HOA_2PI / 3.;
414  azimuths[3] = 2. * HOA_2PI / 3.;
415  elevations[1] = elevations[2] = elevations[3] = HOA_PI2 - HOA_2PI / 3.;
416  }
417  else if(numberOfPoints == 5) // Not regular : Octahedron with only 3 points at 0 azymuth
418  {
419  azimuths[0] = 0.;
420  elevations[0] = HOA_PI2;
421  azimuths[1] = 0.;
422  azimuths[2] = HOA_2PI / 3.;
423  azimuths[3] = 2. * HOA_2PI / 3.;
424  elevations[1] = elevations[2] = elevations[3] = 0;
425  azimuths[4] = 0.;
426  elevations[4] = -HOA_PI2;
427  }
428  else if(numberOfPoints == 6) // Octahedron
429  {
430  azimuths[0] = 0.;
431  elevations[0] = HOA_PI2;
432  azimuths[1] = 0.;
433  azimuths[2] = HOA_PI2;
434  azimuths[3] = 2. * HOA_PI2;
435  azimuths[4] = 3. * HOA_PI2;
436  elevations[1] = elevations[2] = elevations[3] = elevations[4] = 0;
437  azimuths[5] = 0.;
438  elevations[5] = -HOA_PI2;
439  }
440  else if(numberOfPoints == 7) // Not regular : Octahedron with only 5 points at 0 azymuth
441  {
442  azimuths[0] = 0.;
443  elevations[0] = HOA_PI2;
444  azimuths[1] = 0.;
445  azimuths[2] = HOA_2PI / 5.;
446  azimuths[3] = 2. * HOA_2PI / 5.;
447  azimuths[4] = 3. * HOA_2PI / 5.;
448  azimuths[5] = 4. * HOA_2PI / 5.;
449  elevations[1] = elevations[2] = elevations[3] = elevations[4] = elevations[5] = 0;
450  azimuths[6] = 0.;
451  elevations[6] = -HOA_PI2;
452  }
453  else if(numberOfPoints == 8) // Hexahedron or cube
454  {
455  azimuths[0] = azimuths[4] = HOA_PI4;
456  azimuths[1] = azimuths[5] = HOA_PI4 + HOA_PI2;
457  azimuths[2] = azimuths[6] = HOA_PI4 + HOA_PI;
458  azimuths[3] = azimuths[7] = HOA_PI4 + HOA_PI + HOA_PI2;
459  elevations[0] = elevations[1] = elevations[2] = elevations[3] = HOA_PI4;
460  elevations[4] = elevations[5] = elevations[6] = elevations[7] = -HOA_PI4;
461  }
462  else if(numberOfPoints == 9) // Not regular : Hexahedron or cube with 1 at PI/2 elevation
463  {
464  azimuths[0] = azimuths[4] = HOA_PI4;
465  azimuths[1] = azimuths[5] = HOA_PI4 + HOA_PI2;
466  azimuths[2] = azimuths[6] = HOA_PI4 + HOA_PI;
467  azimuths[3] = azimuths[7] = HOA_PI4 + HOA_PI + HOA_PI2;
468  elevations[0] = elevations[1] = elevations[2] = elevations[3] = HOA_PI4;
469  elevations[4] = elevations[5] = elevations[6] = elevations[7] = -HOA_PI4;
470  elevations[8] = HOA_PI2;
471  azimuths[8] = 0;
472  }
473  else if(numberOfPoints == 10) // Not regular : Hexahedron or cube with 1 at PI/2 elevation and 1 at -Pi/2 elevation
474  {
475  azimuths[0] = azimuths[4] = HOA_PI4;
476  azimuths[1] = azimuths[5] = HOA_PI4 + HOA_PI2;
477  azimuths[2] = azimuths[6] = HOA_PI4 + HOA_PI;
478  azimuths[3] = azimuths[7] = HOA_PI4 + HOA_PI + HOA_PI2;
479  elevations[0] = elevations[1] = elevations[2] = elevations[3] = HOA_PI4;
480  elevations[4] = elevations[5] = elevations[6] = elevations[7] = -HOA_PI4;
481  elevations[8] = HOA_PI2;
482  elevations[9] = -HOA_PI2;
483  azimuths[9] = azimuths[8] = 0;
484  }
485  else if(numberOfPoints == 11) // Not regular : Hexahedron or cube with 1 at PI/2 elevation and 1 at -Pi/2 elevation
486  {
487  azimuths[0] = azimuths[4] = HOA_PI4;
488  azimuths[1] = azimuths[5] = HOA_PI4 + HOA_PI2;
489  azimuths[2] = azimuths[6] = HOA_PI4 + HOA_PI;
490  azimuths[3] = azimuths[7] = HOA_PI4 + HOA_PI + HOA_PI2;
491  elevations[0] = elevations[1] = elevations[2] = elevations[3] = HOA_PI4;
492  elevations[4] = elevations[5] = elevations[6] = elevations[7] = -HOA_PI4;
493  elevations[8] = HOA_PI2;
494  elevations[9] = elevations[9] = -HOA_PI2;
495  azimuths[9] = azimuths[8] = 0;
496  }
