87  inline double associated_legendre(int band, int argument, double x)
88  {
89  band = abs(band);
90  argument = abs(argument);
91 
92  if(band == argument)
93  {
94  return pow(-1.0f, float(argument)) * double_factorial(2. * argument - 1) * pow(1. - x * x, 0.5 * argument);
95  }
96  else if(band == argument + 1)
97  {
98  return x * (2 * argument + 1) * associated_legendre(argument, argument, x);
99  }
100  else
101  {
102  return ((double)(2 * band - 1) * x * associated_legendre(band - 1, argument, x) - (double)(band + argument - 1.) * associated_legendre(band - 2, argument, x)) / (double)(band - argument);
103  }
104  }

121  inline double spherical_harmonic_elevation(int band, int argument, double elevation) // With l = band and m the argument
122  {
123  // Yel(l, m)(x, y) = sqrt(((2l + 1) / 4Pi) * (l - m)! / (l +m)!) P(l, m)(cos(elevation))
124  return sqrt((double)(2 * band + 1) / (4 * CICM_PI) * (double)factorial(band - argument) / (double)factorial(band + argument)) * associated_legendre(band, argument, cos(elevation));
