170  {
171  l = abs(l);
172  m = abs(m);
173 
174  if(l == m)
175  {
176  return pow(-1.0f, (double)m) * pow(1. - x * x, 0.5 * m) * double_factorial(2. * m - 1);

185  }

524  inline void vector_sort(unsigned int size, double* vector)
525  {
526  int index;
527  double* temp = new double[size];
528  if(temp && size)
529  {
530  index = 0;
531  temp[0] = vector[0];
532  temp[size - 1] = vector[0];
533 
534  for(unsigned int i = 1; i < size; i++)
535  {
536  if(vector[i] < temp[0]) // Get the minimum
537  {
538  temp[0] = vector[i];
539  index = i;
540  }
541  if(vector[i] > temp[size - 1]) // Get the maximum
542  temp[size - 1] = vector[i];
543  }
544  vector[index] -= 1;
545 
546  for(unsigned int i = 1; i < size - 1; i++)
547  {
548  temp[i] = temp[size - 1];
549  index = -1;
550  for(unsigned int j = 0; j < size; j++)
551  {
552  if(vector[j] >= temp[i-1] && vector[j] <= temp[i])
553  {
554  temp[i] = vector[j];
555  index = j;
556  }
557  }
558  if(index > -1)
559  {
560  vector[index] -= 1;
561  }
562  }
563  memcpy(vector, temp, size * sizeof(double));
